🎓 PATRONES IMPLEMENTADOS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1️⃣ SINGLETON - Instancia Única
   ├─ Logger()              → Registrar eventos y logs
   ├─ ConfigurationManager() → Gestionar configuración global
   └─ CacheManager()        → Cache global compartido
   
   Uso: Logger().log('mensaje');

2️⃣ FACTORY METHOD - Crear por Tipo
   ├─ ProductoFactory.crearProducto()
   ├─ Tipos: 'premium', 'estandar', 'economico'
   └─ También: crearProductoDesdeTelefono()
   
   Uso: ProductoFactory.crearProducto(tipo: 'premium', ...)

3️⃣ ABSTRACT FACTORY - Familias de Productos
   ├─ MarcaFactoryGenerator.obtenerFactory('apple')
   ├─ AppleFactory, SamsungFactory, XiaomiFactory
   ├─ Cada factory crea: Tienda + ServicioTecnico
   └─ Garantiza coherencia de marca
   
   Uso: MarcaFactoryGenerator.obtenerFactory('apple')
           .crearTienda('New York')

4️⃣ BUILDER - Construcción Paso a Paso
   ├─ TelefonoBuscadorBuilder  → Búsquedas complejas
   ├─ TelefonoPersonalizadoBuilder → Crear teléfonos
   ├─ Interfaz fluida (chainable)
   └─ Validación en build()
   
   Uso: TelefonoBuscadorBuilder()
        .conMarca('Samsung')
        .conPrecioMaximo(500)
        .conGarantia(true)
        .build()

5️⃣ PROTOTYPE - Clonar Objetos
   ├─ ConfiguracionTelefonoPrototype → Clonable
   ├─ MarcaPrototype → Clonable
   ├─ CarritoPrototype → Clonable
   ├─ PrototypeRegistry → Registro de prototipos
   └─ clone() + cloneConCambios()
   
   Uso: configBase.cloneConCambios(newNombre: 'Nuevo Nombre')

OPCIÓN 2: Usar Patrones en Tu Código
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
import 'creational_patterns/factory_method_pattern.dart';

class MiPantalla extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    final producto = ProductoFactory.crearProducto(
      tipo: 'premium',
      nombre: 'iPhone 15 Pro',
      precio: 1299.99,
    );
    
    return Column(
      children: [
        Text(producto.nombre),
        Text('\$${producto.precio}'),
      ],
    );
  }
}
